16 /**
17  * @brief Génère une tournée aléatoire en visitant chaque sommet exactement une fois.
18  *
19  * La fonction remplit le tableau bestTour avec l'ordre des IDs des villes
20  * et calcule la distance totale de la tournée dans bestDist.
21  *
22  * @param tour Tableau des instances représentant les villes.
23  * @param dist Fonction de distance entre deux villes (eucl2d, att, geo, etc.).
24  * @param bestTour Tableau d'entiers où stocker l'ordre des IDs des villes.
25  * @param bestDist Pointeur où stocker la distance totale calculée.
26  * @param lenght Nombre de villes dans la tournée.
27  */
28 void random_walk(void ** tour, DistanceFuncGenerique dist, int *bestTour, double *bestDist, int lenght)
29 {
30  int n = lenght;
31  if (!tour || !dist || !bestTour || !bestDist || n <= 0){
32  fprintf(stderr,"nn : Paramètres invalides\n");
33  return;
34  }
35 
36  // Tableau des villes disponibles
37  int *dispo = malloc(sizeof(int) * n);
38  if (!dispo) return;
39  for (int i = 0; i < n; i++)
40  dispo[i] = i;
41 
42  srand(time(NULL)); // initialisation du générateur aléatoire
43 
44  // Crée la tournée aléatoire
45  void ** randomTour = malloc(sizeof(void *)*n);
46  int remaining = n; // nombre de villes encore disponibles
47 
48  for (int i = 0; i < n; i++) {
49  int r;
50  if (remaining == 1) {
51  r = 0; // si une seule ville restante, on la prend
52  } else {
53  r = rand() % remaining; // indice aléatoire parmi les restantes
54  }
55 
56  int chosen_index = dispo[r];
57  randomTour[i] = tour[chosen_index];
58  bestTour[i] = chosen_index + 1; // génération du tour avec indexation à +1
59  dispo[r] = dispo[remaining - 1]; // remplacer l'indice choisi par le dernier disponible
60  remaining--;
61  }
62 
63  // Calcul de la distance totale (boucle fermée)
64  double totalDist = 0.0;
65  for (int i = 0; i < n - 1; i++) {
66  void * a = randomTour[i];
67  void * b = randomTour[i+1];
68  totalDist += dist(a, b);
69  }
70 
71  // Retour au point de départ
72  void * last = randomTour[n-1];
73  void * first = randomTour[0];
74  totalDist += dist(last, first);
75 
76  // Stocker les résultats
77  *bestDist = totalDist;
78 
79  free(dispo);
80  free(randomTour);
81 }
