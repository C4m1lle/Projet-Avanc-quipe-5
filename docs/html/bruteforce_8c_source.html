14 /**
15  * @brief Génère la permutation suivante dans l'ordre lexicographique.
16  *
17  * Cette fonction modifie le tableau en place pour obtenir la permutation suivante.
18  *
19  * @param ids Tableau d'entiers représentant une permutation.
20  * @param length Taille du tableau.
21  * @return 0 si une nouvelle permutation a été générée, 1 si la dernière permutation est atteinte.
22  */
23 int next_permutation(int ids[], int length){
24  // Find non-increasing suffix
25  if (length == 0)
26  return 1;
27  int i = length - 1;
28  while (i > 0 && ids[i - 1] >= ids[i])
29  i--;
30  if (i == 0)
31  return 1;
32 
33  // Find successor to pivot
34  int j = length - 1;
35  while (ids[j] <= ids[i - 1])
36  j--;
37  int temp = ids[i - 1];
38  ids[i - 1] = ids[j];
39  ids[j] = temp;
40 
41  // Reverse suffix
42  j = length - 1;
43  while (i < j) {
44  temp = ids[i];
45  ids[i] = ids[j];
46  ids[j] = temp;
47  i++;
48  j--;
49  }
50  return 0;
51 }

67  void * inst1;
68  void *inst2;

72  int * tab_id = malloc(sizeof(int)*taille);
73 
74  for(int i = 0;i<taille;i++){
75  tab_id[i] = i+1;
76  }
77 
78  while(next_permutation(tab_id,taille)==0 && (*has_to_stop)==0){
79  distcur = 0;
80  for(int i = 0;i<taille-1;i++){
81  inst1 = tour[tab_id[i]-1];
82  inst2 = tour[tab_id[i+1]-1];
83  distcur+=dist(inst1,inst2);
84  }
85  distcur+=dist(tour[tab_id[0]-1],tour[tab_id[taille-1]-1]);
86 
87  if((*distmin)>distcur){
88  (*distmin) = distcur;
89  for(int i = 0;i<taille;i++){
90  best[i] = tab_id[i];
91  }
92  }
93  }
94  free(tab_id);
95 }
