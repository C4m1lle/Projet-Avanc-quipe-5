16 /**
17  * @brief Vérifie si deux segments sont croisés.
18  *
19  * @param A Première extrémité du premier segment.
20  * @param B Deuxième extrémité du premier segment.
21  * @param C Première extrémité du second segment.
22  * @param D Deuxième extrémité du second segment.
23  * @return 1 si les segments se croisent, 0 sinon.
24  */
25 int is_crossed(tInstance A, tInstance B, tInstance C, tInstance D){
26  double ax = get_x(A), ay = get_y(A);
27  double bx = get_x(B), by = get_y(B);
28  double cx = get_x(C), cy = get_y(C);
29  double dx = get_x(D), dy = get_y(D);
30 
31  double d1, d2, d3, d4;
32 
33  // produits vectoriels
34  d1 = (bx - ax)*(cy - ay) - (by - ay)*(cx - ax);
35  d2 = (bx - ax)*(dy - ay) - (by - ay)*(dx - ax);
36  d3 = (dx - cx)*(ay - cy) - (dy - cy)*(ax - cx);
37  d4 = (dx - cx)*(by - cy) - (dy - cy)*(bx - cx);
38 
39  // segments croisés si les signes sont opposés
40  if((d1 * d2 < 0) && (d3 * d4 < 0)){
41  return 1;
42  }
43  return 0;
44 }
