31 tDemiMatrice creer_demi_matrice(int nb_villes) {
32  tDemiMatrice matrice = malloc(sizeof(struct sDemiMatrice));
33  if (!matrice) return NULL;
34 
35  matrice->nb_villes = nb_villes;
36  matrice->lignes = malloc(nb_villes * sizeof(double*));
37  if (!matrice->lignes) { free(matrice); return NULL; }
38 
39  for (int i = 0; i < nb_villes - 1; i++) {
40  matrice->lignes[i] = malloc((nb_villes - i - 1) * sizeof(double));
41  if (!matrice->lignes[i]) {
42  for (int k = 0; k < i; k++) free(matrice->lignes[k]);
43  free(matrice->lignes);
44  free(matrice);
45  return NULL;
46  }
47  }
48  matrice->lignes[nb_villes - 1] = NULL;
49  return matrice;
50 }

64 /**
65  * @brief Obtient la distance entre deux villes dans une demi-matrice.
66  *
67  * @param matrice Demi-matrice contenant les distances.
68  * @param i Index de la première ville.
69  * @param j Index de la deuxième ville.
70  * @return Distance entre les deux villes, ou -1.0 si indices invalides, 0.0 si i == j.
71  */
72 double obtenir_distance(const tDemiMatrice matrice, int i, int j) {
73  if (!matrice || i < 0 || j < 0 || i >= matrice->nb_villes || j >= matrice->nb_villes) return -1.0;
74  if (i == j) return 0.0;
75  return (i < j) ? matrice->lignes[i][j-i-1] : matrice->lignes[j][i-j-1];
76 }

78 /**
79  * @brief Définit la distance entre deux villes dans une demi-matrice.
80  *
81  * @param matrice Demi-matrice contenant les distances.
82  * @param i Index de la première ville.
83  * @param j Index de la deuxième ville.
84  * @param valeur Valeur de la distance à définir.
85  * @return 0 si succès, -1 si indices invalides, 0 si i == j.
86  */
87 int definir_distance(tDemiMatrice matrice, int i, int j, double valeur) {
88  if (!matrice || i < 0 || j < 0 || i >= matrice->nb_villes || j >= matrice->nb_villes) return -1;
89  if (i == j) return 0;
90  if (i < j) matrice->lignes[i][j-i-1] = valeur;
91  else matrice->lignes[j][i-j-1] = valeur;
92  return 0;
93 }

97  *
98  * @param tour Tableau de pointeurs vers les villes.
99  * @param nb_villes Nombre de villes.

102  */
103 tDemiMatrice demi_matrice_from_tour(void** tour, int nb_villes, DistanceFuncGenerique dist) {
104  if (!tour || !dist || nb_villes <= 0) return NULL;
105 
106  tDemiMatrice matrice = creer_demi_matrice(nb_villes);
107  if (!matrice) return NULL;
108 
109  for (int i = 0; i < nb_villes - 1; i++) {
110  void * inst_i = tour[i];
111  for (int j = i + 1; j < nb_villes; j++) {
112  void * inst_j = tour[j];
113  double d = dist(inst_i, inst_j);
114  definir_distance(matrice, i, j, d);
115  }
116  }
117 
118  return matrice;
119 }
