165 /**
166  * @brief Inverse un segment d'une tournée entre deux indices.
167  * @param tournee Tournée
168  * @param id1 Index de départ
169  * @param id2 Index de fin
170  * @return int 0 si succès, -1 si indices invalides
171  */
172 int reverse_segment(tTournee tournee, int id1, int id2){
173  if(id1 > tournee->current-1 || id2 > tournee->current-1){
174  return -1;
175  }
176  tInstance tmp;
177  while (id1 < id2) {
178  tmp = tournee->chemin[id1];
179  tournee->chemin[id1] = tournee->chemin[id2];
180  tournee->chemin[id2] = tmp;
181  id1++;
182  id2--;
183  }
184  return 0;
185 }
