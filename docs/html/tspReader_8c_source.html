70  * @brief Charge un problème TSP depuis un fichier TSPLIB.
71  * @param filepath Chemin vers le fichier .tsp
72  * @return tProbleme Problème chargé, NULL en cas d'erreur
73  */
74 tProbleme load_problem(const char * filepath){
75  char token_buffer[128];
76  char c;
77  tInstance current_instance;
78  int i,j;
79  int sum;
80  int id;
81  double x,y;
82 
83  /* Ouverture du fichier */
84  if((pfile = fopen(filepath,"r")) == NULL){
85  fprintf(stderr,"Fichier %s inaccessible ou inexistant.\n",filepath);
86  return NULL;
87  }
88  tProbleme problem = create_problem();
89 
90  /* Lecture du fichier TSPLIB */
91  do{
92  j=0;
93  do{
94  c = fgetc(pfile);
95  token_buffer[j]=c;
96  j++;
97  }while(c != ' ' && c!= ':' && j<127 && c != '\n');
98  token_buffer[j-1]='\0';
99  if(c == ' '){
100  while(fgetc(pfile)!=':');
101  }
102 
103  /* Calcul de la somme ASCII du mot-clé */
104  i = 0;
105  sum = 0;
106  while(token_buffer[i]!='\0' && i < 128){
107  sum+=token_buffer[i];
108  i++;
109  }
110 
111  switch(sum){
112  case NAME:
113  fscanf(pfile,"%s",token_buffer);
114  problem->name = string_alloc(token_buffer);
115  break;
116  case COMMENT:
117  fscanf(pfile,"%[^\n]",token_buffer);
118  problem->comment = string_alloc(token_buffer);
119  break;
120  case TYPE:
121  fscanf(pfile,"%s",token_buffer);
122  problem->type = string_alloc(token_buffer);
123  break;
124  case DIMENSION:
125  fscanf(pfile,"%s",token_buffer);
126  problem->dimension = atoi(token_buffer);
127  break;
128  case EDGE_WEIGHT_TYPE:
129  fscanf(pfile,"%s",token_buffer);
130  problem->edge_weight_type = string_alloc(token_buffer);
131  break;
132  case NODE_COORD_SECTION:
133  problem->tournee = create_tournee(problem->dimension);
134  for(int i = 0; i<problem->dimension;i++){
135  fscanf(pfile,"%d %lf %lf",&id,&x,&y);
136  current_instance = create_instance(id,x,y);
137  add_in_tournee(problem->tournee,current_instance);
138  }
139  break;
140  case DISPLAY_DATA_TYPE:
141  fscanf(pfile,"%s",token_buffer);
142  problem->display_data_type = string_alloc(token_buffer);
143  break;
144  case 0: // mot vide
145  break;
146  case 218: // EOF
147  break;
148  default:
149  fprintf(stderr,"Erreur : Mot clé invalide dans %s, \"%s\" non traité\n",filepath,token_buffer);
150  break;
151  }
152 
153  }while(strcmp(token_buffer,"EOF")!=0);
154 
155  return problem;
156 }
