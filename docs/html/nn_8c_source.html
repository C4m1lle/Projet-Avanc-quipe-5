15 /**
16  * @brief Heuristique du plus proche voisin (Nearest Neighbor).
17  *
18  * Cette fonction construit une tournée en choisissant à chaque étape la ville
19  * la plus proche parmi celles qui n'ont pas encore été visitées.
20  *
21  * @param tour Tableau des instances représentant les villes.
22  * @param distance Fonction de calcul de distance entre deux instances.
23  * @param meilleure_tour Tableau résultat contenant l'ordre des villes (indexés à partir de 1).
24  * @param longueur_totale Longueur totale finale de la tournée construite.
25  * @param lenght Nombre de villes dans la tournée.
26  */
27 void plus_proche_voisin(void ** tour, DistanceFuncGenerique distance, int *meilleure_tour, double *longueur_totale,int lenght)
28 {
29  int n = lenght;
30  if (n <= 0 || meilleure_tour == NULL) {
31  return;
32  }
33 
34  int *visite = malloc(n * sizeof(int));
35  if (visite == NULL) {
36  fprintf(stderr, "Erreur : allocation mémoire\n");
37  return;
38  }
39 
40  int i, j;
41  for (i = 0; i < n; i++) {
42  visite[i] = 0;
43  }
44 
45  int ville_courante = 0;
46  visite[ville_courante] = 1;
47  meilleure_tour[0] = ville_courante + 1;
48 
49  for (i = 1; i < n; i++) {
50  double distance_min = DBL_MAX;
51  int prochaine_ville = -1;
52 
53  void * instance_actuelle = tour[ville_courante];
54 
55  for (j = 0; j < n; j++) {
56  if (visite[j] == 0) {
57  void * instance_cand = tour[j];
58  double d = distance(instance_actuelle, instance_cand);
59  if (d < distance_min) {
60  distance_min = d;
61  prochaine_ville = j;
62  }
63  }
64  }
65 
66  if (prochaine_ville == -1) {
67  for (j = i; j < n; j++) {
68  meilleure_tour[j] = -1;
69  }
70  i = n;
71  } else {
72  meilleure_tour[i] = prochaine_ville + 1;
73  visite[prochaine_ville] = 1;
74  ville_courante = prochaine_ville;
75  }
76  }
77 
78  double somme_distances = 0.0;
79  for (i = 0; i < n - 1; i++) {
80  void * a = tour[meilleure_tour[i]-1];
81  void * b = tour[meilleure_tour[i + 1]-1];
82  somme_distances += distance(a, b);
83  }
84 
85  void * debut = tour[meilleure_tour[0]-1];
86  void * fin = tour[meilleure_tour[n - 1]-1];
87  somme_distances += distance(fin, debut);
88 
89  *longueur_totale = somme_distances;
90 
91  free(visite);
92 }
