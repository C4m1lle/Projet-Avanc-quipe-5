316  void *data = params->data;
317 
318  srand((unsigned int)time(NULL));
319 
320  /* Best = premier individu */
321  Individual best = copy(population[0]);
322  double best_len = fitness(best, data);
323 
324  /* BOUCLE GA */
325  for (int g = 0; g < generations; g++) {
326 
327  /* 1) SELECTION */
328  Individual *selected = malloc(sizeof(Individual) * pop_size);
329  for (int i = 0; i < pop_size; i++)
330  selected[i] = selection(population, pop_size, tournament_sz, fitness, data);
331 
332  /* 2) CROSSOVER */
333  Individual *offspring = malloc(sizeof(Individual) * pop_size);
334  for (int i = 0; i < pop_size; i += 2) {
335  offspring[i] = crossover(selected[i], selected[i+1], data);
336  offspring[i+1] = crossover(selected[i+1], selected[i], data);
337  }
338 
339  /* 3) MUTATION */
340  for (int i = 0; i < pop_size; i++)
341  mutate(offspring[i], mutation_r);
342 
343  /* 4) TRI */
344  for (int i = 0; i < pop_size - 1; i++)
345  for (int j = i + 1; j < pop_size; j++)
346  if (fitness(offspring[i], data) > fitness(offspring[j], data)) {
347  Individual tmp = offspring[i];
348  offspring[i] = offspring[j];
349  offspring[j] = tmp;
350  }
351 
352  /* 5) REMPLACER LE PIRE PAR RANDOM TOUR */
353  int worst = pop_size - 1;
354  del(offspring[worst]);
355  offspring[worst] = create_random(individual_sz, data);
356 
357  /* 6) GARDER LE MEILLEUR GLOBAL */
358  if (fitness(offspring[0], data) < best_len) {
359  del(best);
360  best = copy(offspring[0]);
361  best_len = fitness(best, data);
362  }
363 
364  /* 7) remplacer par le meilleur individu */
365  del(offspring[worst]);
366  offspring[worst] = copy(best);
367 
368  /* 8) Remplacement de la population */
369  for (int i = 0; i < pop_size; i++) del(population[i]);
370  free(population);
371  population = offspring;
372 
373  /* Libération de selected */
374  for (int i = 0; i < pop_size; i++) del(selected[i]);
375  free(selected);
376  }
377 
378  /* Retour des résultats */
379  *best_ind = best;
380  *best_score = best_len;
381 
382  /* Nettoyage */
383  for (int i = 0; i < pop_size; i++) del(population[i]);
384  free(population);
385 
386  return 0;
